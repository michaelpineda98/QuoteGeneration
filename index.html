#!/usr/bin/env python3 """ generate_quote_tool.py Usage: python generate_quote_tool.py [path/to/_MP_Items.xlsx] [output.html] Reads _MP_Items.xlsx (sheet: _MP_Items): Column 1 = Part Number Column 2 = Description Column 4 = Unit Cost -> Part Cost = round(unit_cost * 225 / 100, 2) """ import sys import json import pandas as pd from pathlib import Path def load_parts(xlsx_path: str) -> list[dict]: df = pd.read_excel(xlsx_path, sheet_name="_MP_Items", header=0) try: col_pn = df.columns[0] col_desc = df.columns[1] col_unit = df.columns[3] col_curr = df.columns[4] except IndexError: print(f"ERROR: Expected at least 5 columns, got {len(df.columns)}: {list(df.columns)}") sys.exit(1) print(f" Columns detected:") print(f" [1] Part Number -> '{col_pn}'") print(f" [2] Description -> '{col_desc}'") print(f" [4] Unit Cost -> '{col_unit}'") print(f" [5] Curr. Unit Cost -> '{col_curr}'") parts = [] for _, row in df.iterrows(): pn = str(row[col_pn]).strip() if not pn or pn.lower() in ("nan", "none", ""): continue desc = str(row[col_desc]).strip() if pd.notna(row[col_desc]) else "" if desc.lower() == "nan": desc = "" try: unit_cost = float(row[col_unit]) except (ValueError, TypeError): unit_cost = 0.0 try: curr_unit_cost = float(row[col_curr]) except (ValueError, TypeError): curr_unit_cost = unit_cost part_cost = round(unit_cost * 225 / 100, 2) parts.append({ "part_number": pn, "description": desc, "unit_cost": round(unit_cost, 2), "curr_unit_cost": round(curr_unit_cost, 2), "part_cost": part_cost, }) return parts def generate_html(parts: list[dict], output_path: str): parts_json = json.dumps(parts) # NOTE: Plain string — NO f-string.
